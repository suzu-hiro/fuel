Gradation TOP

基本

.item {
  background: linear-gradient(#ff9999, #99ff99);
}

方向

.item {
  background: linear-gradient(to right, #ff9999, #99ff99);
}

対角グラデーション

.item {
  background: linear-gradient(to bottom right, #ff9999, #99ff99);
}

角度指定

.item {
  background: linear-gradient(30deg, #ff9999, #99ff99);
}

3色以上

.item {
  background: linear-gradient(90deg, #ff9999, #99ff99, #9999ff);
}

色の配置

.item {
  background: linear-gradient(
    90deg, #ff9999 0%, #99ff99 30%, #9999ff 50%
  );
}

不連続線

.item {
  background: linear-gradient(
    90deg, #ff9999 30%, #99ff99 30%, #99ff99 70%, #9999ff 70%
  );
}

反復

.item {
  background: repeating-linear-gradient(
    90deg, #ff9999, #ff9999 10px, #99ff99 10px, #99ff99 20px
  );
}
